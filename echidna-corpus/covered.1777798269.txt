   9 |     |  * @dev Contract module that helps prevent reentrant calls to a function.
  10 |     |  *
  11 |     |  * Inheriting from `ReentrancyGuard` will make the {nonReentrant} modifier
  12 |     |  * available, which can be applied to functions to make sure there are no nested
  13 |     |  * (reentrant) calls to them.

  16 |     |  * `nonReentrant` may not call one another. This can be worked around by making
  17 |     |  * those functions `private`, and then adding `external` `nonReentrant` entry
  18 |     |  * points to them.

  21 |     |  * consider using {ReentrancyGuardTransient} instead.

  23 |     |  *            node calls `performUpkeep()` to transfer the escrowed ETH to the backup

 417 |     |     /**
 418 |     |      * @notice Determines which wallets (if any) are ready for recovery.
 419 |     |      *
 420 |     |      * @dev    Called off-chain by Chainlink Automation nodes. Gas cost is not a
 421 |     |      *         concern here; the function iterates a paginated window of wallets.
 422 |     |      *
 423 |     |      *         Pagination via `checkData`:
 424 |     |      *           `checkData = abi.encode(uint256 startIndex, uint256 batchSize)`
 425 |     |      *         If `checkData` is empty, defaults to startIndex=0, batchSize=MAX_BATCH_SIZE.
 426 |     |      *         Callers should paginate across the full registry in separate upkeep
 427 |     |      *         registrations when the wallet count is large.
 428 |     |      *
 429 |     |      * @param  checkData ABI-encoded (startIndex, batchSize) for pagination.
 430 |     |      * @return upkeepNeeded True if at least one wallet is due for recovery.
 431 |     |      * @return performData  ABI-encoded address[] of wallets ready for recovery.
 432 |     |      */
 433 |     |     function checkUpkeep(bytes calldata checkData)
 434 |     |         external
 435 |     |         view
 436 |     |         override(IAeternumVault, AutomationCompatibleInterface)
 437 |     |         returns (bool upkeepNeeded, bytes memory performData)
 438 |     |     {
 439 |     |         uint256 startIndex;
 440 |     |         uint256 batchSize;
 441 |     | 
 442 |     |         if (checkData.length > 0) {
 443 |     |             (startIndex, batchSize) = abi.decode(checkData, (uint256, uint256));
 444 |     |         } else {
 445 |     |             startIndex = 0;
 446 |     |             batchSize = MAX_BATCH_SIZE;
 447 |     |         }
 448 |     | 
 449 |     |         // Cap batchSize to MAX_BATCH_SIZE to mirror performUpkeep limits
 450 |     |         if (batchSize > MAX_BATCH_SIZE) {
 451 |     |             batchSize = MAX_BATCH_SIZE;
 452 |     |         }
 453 |     | 
 454 |     |         uint256 totalWallets = s_registeredWallets.length;
 455 |     | 
 456 |     |         // Early exit: nothing to check
 457 |     |         if (totalWallets == 0 || startIndex >= totalWallets) {
 458 |     |             return (false, bytes(""));
 459 |     |         }
 460 |     | 
 461 |     |         uint256 endIndex = startIndex + batchSize;
 462 |     |         if (endIndex > totalWallets) endIndex = totalWallets;
 463 |     | 
 464 |     |         // Allocate maximum possible size; trim with assembly after filling
 465 |     |         address[] memory candidates = new address[](endIndex - startIndex);
 466 |     |         uint256 count = 0;
 467 |     | 
 468 |     |         for (uint256 i = startIndex; i < endIndex;) {
 469 |     |             address wallet = s_registeredWallets[i];
 470 |     |             RecoveryConfig storage config = s_configs[wallet];
 471 |     | 
 472 |     |             if (
 473 |     |                 config.isActive && config.balance > 0
 474 |     |                     && block.timestamp >= config.lastActivity + config.inactivityPeriod
 475 |     |             ) {
 476 |     |                 candidates[count] = wallet;
 477 |     |                 unchecked {
 478 |     |                     ++count;
 479 |     |                 }
 480 |     |             }
 481 |     | 
 482 |     |             unchecked {
 483 |     |                 ++i;
 484 |     |             }
 485 |     |         }
 486 |     | 
 487 |     |         if (count > 0) {
 488 |     |             // Trim the array to the actual number of candidates found
 489 |     |             assembly {
 490 |     |                 mstore(candidates, count)
 491 |     |             }
 492 |     |             upkeepNeeded = true;
 493 |     |             performData = abi.encode(candidates);
 494 |     |         }
 495 |     |     }

 658 |     |     /**
 659 |     |      * @notice Executes recovery for a single wallet, transferring its balance to the backup address.
 660 |     |      * @dev Attempts to execute recovery for a single wallet.
 661 |     |      *      Re-validates all conditions so stale or duplicate entries are safely skipped.
 662 |     |      *
 663 |     |      *      CEI pattern:
 664 |     |      *        1. Checks  — re-validate eligibility.
 665 |     |      *        2. Effects — zero the balance, mark inactive, remove from registry.
 666 |     |      *        3. Interact — transfer ETH to backupAddress.
 667 |     |      *
 668 |     |      *      Failure handling & retry logic:
 669 |     |      *        * If the ETH transfer to `backupAddress` fails (e.g. reverting contract),
 670 |     |      *          state is safely restored and a {RecoveryFailed} event is emitted.
 671 |     |      *        * The wallet is re-added to the monitoring queue for retry in the next upkeep cycle.
 672 |     |      *        * Recovery is retried up to `MAX_RECOVERY_ATTEMPTS` (currently 3).
 673 |     |      *        * Once the max attempts threshold is reached, the wallet is marked as abandoned,
 674 |     |      *          permanently removed from the monitoring queue, and {RecoveryAbandoned} is emitted.
 675 |     |      *        * Wallet balance remains accessible via withdrawAll() or send(),
 676 |     |      *          but re-registration with same backupAddress is permanently blocked.
 677 |     |      *
 678 |     |      *      This design ensures that:
 679 |     |      *        * A single faulty backup address cannot block batch execution.
 680 |     |      *        * Recovery attempts are bounded, preventing infinite retry loops.
 681 |     |      *
 682 |     |      *      Slither reentrancy-eth: acknowledged.
 683 |     |      *        * On a failed transfer, config.balance, config.isActive, config.isAbandoned,
 684 |     |      *          and s_registeredWallets are written after the external call — unavoidable
 685 |     |      *          in the failure-restore pattern.
 686 |     |      *        * Safety is guaranteed by three independent layers:
 687 |     |      *            1. `nonReentrant` on `performUpkeep` blocks all reentrant calls.
 688 |     |      *            2. `config.balance` is zeroed before the call — no double-spend possible.
 689 |     |      *            3. After MAX_RECOVERY_ATTEMPTS (3) consecutive failures the vault is
 690 |     |      *               permanently abandoned, eliminating the retry loop entirely.
 691 |     |      *
 692 |     |      * @param wallet The wallet address to attempt recovery on.
 693 |     |      */
 694 | *   |     function _executeRecovery(address wallet) internal {
 695 | *   |         RecoveryConfig storage config = s_configs[wallet];
 696 |     | 
 697 |     |         // Re-validate (Checks)
 698 | *   |         if (!config.isActive) return;
 699 |     |         if (config.balance == 0) return;
 700 |     |         if (block.timestamp < config.lastActivity + config.inactivityPeriod) return;
 701 |     | 
 702 |     |         address backupAddress = config.backupAddress;
 703 |     |         uint256 amount = config.balance;
 704 |     | 
 705 |     |         // Pre-calculate new attempt count before external call (CEI)
 706 |     |         uint8 newAttempts;
 707 |     |         unchecked {
 708 |     |             newAttempts = config.failedRecoveryAttempts + 1;
 709 |     |         }
 710 |     | 
 711 |     |         // Effects — ALL moved before external call
 712 |     |         config.balance = 0;
 713 |     |         config.isActive = false;
 714 |     |         config.failedRecoveryAttempts = newAttempts;
 715 |     |         _removeFromRegistry(wallet);
 716 |     | 
 717 |     |         // Interaction
 718 |     |         // slither-disable-next-line reentrancy-eth
 719 |     |         (bool success,) = backupAddress.call{value: amount}("");
 720 |     | 
 721 |     |         if (!success) {
 722 |     |             // Restore balance — safe: nonReentrant on performUpkeep prevents exploitation.
 723 |     |             // config.balance was zeroed before the call so no double-spend is possible.
 724 |     |             config.balance = amount;
 725 |     | 
 726 |     |             if (newAttempts >= MAX_RECOVERY_ATTEMPTS) {
 727 |     |                 config.isAbandoned = true;
 728 |     |                 emit RecoveryAbandoned(wallet, backupAddress, amount);
 729 |     |                 return;
 730 |     |             }

 732 |     |             // Restore monitoring state for retry next cycle
 733 |     |             config.isActive = true;
 734 |     |             s_registeredWallets.push(wallet);
 735 |     |             s_walletIndexPlusOne[wallet] = s_registeredWallets.length;
 736 |     |             emit RecoveryFailed(wallet, backupAddress, amount);
 737 |     |             return;
 738 |     |         }
 739 |     | 
 740 |     |         emit RecoveryExecuted(wallet, backupAddress, amount);
 741 |     |     }

 743 |     |     /**
 744 |     |      * @notice Removes a wallet from the registered wallets array in O(1) using swap-and-pop.
 745 |     |      * @dev Removes `wallet` from `s_registeredWallets`.
 746 |     |      *
 747 |     |      *      Algorithm:
 748 |     |      *        1. Look up the wallet's 1-indexed position.
 749 |     |      *        2. Move the last element in the array into that position.
 750 |     |      *        3. Update the moved element's index mapping.
 751 |     |      *        4. Pop the (now-duplicate) last element and delete the mapping entry.
 752 |     |      *
 753 |     |      *      This preserves array integrity when multiple wallets are removed in the
 754 |     |      *      same `performUpkeep` call, because `s_walletIndexPlusOne` is always
 755 |     |      *      updated to reflect the current (possibly new) position.
 756 |     |      *
 757 |     |      * @param wallet Address to remove from the registry.
 758 |     |      */
 759 |     |     function _removeFromRegistry(address wallet) internal {
 760 |     |         uint256 indexPlusOne = s_walletIndexPlusOne[wallet];
 761 |     |         if (indexPlusOne == 0) return; // Not in registry; nothing to do.
 762 |     | 
 763 |     |         uint256 indexToRemove = indexPlusOne - 1;
 764 |     |         uint256 lastIndex = s_registeredWallets.length - 1;
 765 |     | 
 766 |     |         if (indexToRemove != lastIndex) {
 767 |     |             // Move the last element into the gap
 768 |     |             address lastWallet = s_registeredWallets[lastIndex];
 769 |     |             s_registeredWallets[indexToRemove] = lastWallet;
 770 |     |             s_walletIndexPlusOne[lastWallet] = indexPlusOne; // update moved element's index
 771 |     |         }
 772 |     | 
 773 |     |         s_registeredWallets.pop();
 774 |     |         delete s_walletIndexPlusOne[wallet];
 775 |     |     }

 778 |     |                              FALLBACK / RECEIVE
 779 |     |     //////////////////////////////////////////////////////////////*/
